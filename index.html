function Jumper.OnUpdateHandler( Event, ticks )     
    if ( Jumper.running and Jumper.initialized ) then
        if ( ticks - Jumper.lastticks > 500 ) then
            if ( Jumper.lastticks == 0 ) then
                Jumper.lastticks = ticks + 2000
                return
            end         
            Jumper.lastticks = ticks
        end
    end
end
